    | bool

    | { l_0 = v_0, l_1 = v_1, ... l_n = v_n } (n >= 0)
    | < l_i = v_i >
      : < l_0 : T_0, l_1 : T_1, ... l_i : T_i, ... l_n : T_n >
      (n >= 0)
    | fun x -> e
